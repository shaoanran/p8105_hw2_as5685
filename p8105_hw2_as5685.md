pols_month_data = read_csv("./fivethirtyeight_datasets/pols-month.csv") %>%
  janitor::clean_names() %>%
  separate(mon,into = c("year","month", "day"),sep = "-") %>% 
  pivot_longer(
    cols = starts_with("prez"),
    names_to = "prezsident", 
    names_prefix = "prez_",
    values_to = "value"
  ) %>% 
filter(value != "0") %>% 
select(-day)
